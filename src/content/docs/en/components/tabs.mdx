import { Tabs, TabItem } from '@astrojs/starlight/components';
import { Card, CardGrid } from '@astrojs/starlight/components';

You can display a tabbed interface using the ```<Tabs>``` and ```<TabItem>``` 
components. Each ```<TabItem>``` must have a label to display to users. Use the 
optional icon attribute to include one of Starlight's built-in icons next to the 
label.

<Tabs>
  <TabItem label="Stars" icon="star">
    Sirius, Vega, Betelgeuse
  </TabItem>
  <TabItem label="Moons" icon="moon">
    Io, Europa, Ganymede
  </TabItem>
</Tabs>
