import { Card, CardGrid } from '@astrojs/starlight/components';

You can display content in a box matching Starlight's styles using the 
```<Card>``` component. Wrap multiple cards in the ```<CardGrid>``` component to 
display cards side-by-side when there's enough space.

<Card title="Check this out">Interesting content you want to highlight.</Card>

<CardGrid>
  <Card title="Stars" icon="star">
    Sirius, Vega, Betelgeuse
  </Card>
  <Card title="Moons" icon="moon">
    Io, Europa, Ganymede
  </Card>
</CardGrid>
